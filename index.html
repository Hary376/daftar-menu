# --- CONTOH KODE PYTHON ---

# 1. Variabel
nama_produk = "Kopi Susu"
harga = 15000

# 2. List (Array)
menu = ["Kopi Hitam", "Kopi Susu", "Teh Tarik"]

# 3. Kondisi (If-Else)

    status = "Tersedia"
else:
    status = "Habis"

# 4. Looping
print(f"Status {nama_produk}: {status}")
print("Daftar Menu:")
for item in menu:
    print(f"- {item}")
